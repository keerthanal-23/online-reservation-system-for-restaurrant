    <script>
        $(document).ready(function() {
            // Hide error messages
            $(".error-message").hide();

            // Validate form inputs when submitting
            $("form").submit(function(event) {
                var name = $("#name").val();
                var email = $("#email").val();
                var phone = $("#phone").val();
                var date = $("#date").val();
                var time = $("#time").val();
                var guests = $("#guests").val();

                // Check name
                if (name === "") {
                    $("#name-error").show();
                    event.preventDefault();
                } else {
                    $("#name-error").hide();
                }

                // Check email
                if (email === "") {
                    $("#email-error").show();
                    event.preventDefault();
                } else {
                    $("#email-error").hide();
                }

                // Check phone
                if (phone === "") {
                    $("#phone-error").show();
                    event.preventDefault();
                } else {
                    $("#phone-error").hide();
                }

                // Check date
                if (date === "") {
                    $("#date-error").show();
                    event.preventDefault();
                } else {
                    $("#date-error").hide();
                }

                // Check time
                if (time === "") {
                    $("#time-error").show();
                    event.preventDefault();
                } else {
                    $("#time-error").hide();
                }

                // Check number of guests
                if (guests === "" || guests <= 0) {
                    $("#guests-error").show();
                    event.preventDefault();
                } else {
                    $("#guests-error").hide();
                }
            });

            // Update time options based on date selection
            $("#date").change(function() {
                var date = $(this).val();
                if (date === "") {
                    $("#time option").show();
                } else {
                    $("#time option").each(function() {
                        var time = $(this).val();
                        if (time === "5:00 PM" || time === "5:30 PM" || time === "6:00 PM") {
                            $(this).hide();
                        } else {
                            $(this).show();
                        }
                    });
                }
            });
        });
    </script>
